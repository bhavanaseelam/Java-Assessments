public class Factor{
    public static void printFactors(int num){
        System.out.print("Factors of " + num + " are: ");
        for (int i = 1; i <= num; i++) {
            if (num % i == 0) {
                System.out.print(i + " ");
            }
        }
        System.out.println();
    }

    public static void main(String[] args) {
        printFactors(18); 
        printFactors(45); 
    }
}
